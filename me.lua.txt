-- Lade die Rayfield-Bibliothek
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Erstelle das Hauptfenster mit erweiterten Optionen
local Window = Rayfield:CreateWindow({
    Name = "Ultimate Script Hub",
    LoadingTitle = "Loading Advanced Interface",
    LoadingSubtitle = "Please wait, initializing features...",
    ConfigurationSaving = {
        Enabled = true,
        FolderName = "AdvancedGUI", -- Speicherort für Einstellungen
        FileName = "ScriptHubConfig"
    },
    Discord = {
        Enabled = true,
        Invite = "your-invite-code", -- Füge hier eine Discord-Einladung ein
        RememberJoins = true
    },
    Theme = "Ocean", -- Standard-Theme
    KeySystem = true, -- Aktiviere den Key-System-Schutz
    KeySettings = {
        Title = "Access Key Required",
        Subtitle = "Please enter your key to proceed.",
        Note = "If you don't have a key, contact support.",
        Key = "Your-Key-Here" -- Ersetze mit deinem Schlüssel
    }
})

-- **Tab 1: Hauptfunktionen**
local MainTab = Window:CreateTab("Script Hub", 4483362458)

-- **Script-Button: Führt dein Script aus**
MainTab:CreateButton({
    Name = "Execute Cheese Script",
    Callback = function()
        local success, errorMessage = pcall(function()
            game:GetService("ReplicatedStorage").AddCheese:FireServer()
        end)

        if success then
            Rayfield:Notify({
                Title = "Script Execution",
                Content = "Cheese Script executed successfully!",
                Duration = 5,
                Actions = {
                    Close = { Name = "OK", Callback = function() end }
                }
            })
        else
            Rayfield:Notify({
                Title = "Execution Error",
                Content = "Failed to execute Cheese Script: " .. errorMessage,
                Duration = 5,
                Actions = {
                    Close = { Name = "OK", Callback = function() end }
                }
            })
        end
    end,
})

-- **Slider zur Lautstärkeanpassung**
MainTab:CreateSlider({
    Name = "Adjust Volume",
    Range = {0, 100},
    Increment = 1,
    Suffix = "%",
    CurrentValue = 50,
    Callback = function(Value)
        print("Volume set to:", Value)
    end,
})

-- **Farbwähler mit Akzentänderung**
MainTab:CreateColorPicker({
    Name = "Change Accent Color",
    Color = Color3.fromRGB(0, 255, 255), -- Standardfarbe
    Callback = function(Color)
        Window:ModifyTheme({ Accent = Color })
        Rayfield:Notify({
            Title = "Theme Update",
            Content = "Accent color updated!",
            Duration = 3
        })
    end,
})

-- **Dropdown für Modi**
MainTab:CreateDropdown({
    Name = "Select Mode",
    Options = {"Stealth", "Combat", "Explorer"},
    CurrentOption = "Stealth",
    Callback = function(Option)
        print("Mode selected:", Option)
    end,
})

-- **Toggle zur Aktivierung von Features**
MainTab:CreateToggle({
    Name = "Activate Stealth Mode",
    CurrentValue = false,
    Callback = function(State)
        print("Stealth Mode Active:", State)
    end,
})

-- **Keybind für schnelle Aktionen**
MainTab:CreateKeybind({
    Name = "Quick Script Execute",
    CurrentKeybind = Enum.KeyCode.F,
    HoldToInteract = false,
    Callback = function()
        local success, errorMessage = pcall(function()
            game:GetService("ReplicatedStorage").AddCheese:FireServer()
        end)

        if success then
            print("Cheese Script executed via Keybind!")
        else
            print("Keybind execution error:", errorMessage)
        end
    end,
})

-- **Tab 2: Theme-Optionen**
local ThemeTab = Window:CreateTab("Themes", 4483362458)

-- **Theme-Wechsler**
ThemeTab:CreateDropdown({
    Name = "Pick a Theme",
    Options = {"Default", "Ocean", "Serenity", "AmberGlow", "Amethyst"},
    CurrentOption = "Ocean",
    Callback = function(Theme)
        Window:SetTheme(Theme)
        Rayfield:Notify({
            Title = "Theme Change",
            Content = "Theme updated to: " .. Theme,
            Duration = 3
        })
    end,
})

-- **Info-Paragraph**
ThemeTab:CreateParagraph({
    Title = "About",
    Content = "This GUI showcases the ultimate features of Rayfield. Customize your experience and enjoy the versatility."
})

-- **Tab 3: Informationen und Credits**
local InfoTab = Window:CreateTab("Information", 4483362458)

-- **Credits anzeigen**
InfoTab:CreateLabel("Created by Souhail")
InfoTab:CreateParagraph({
    Title = "Special Thanks",
    Content = "Thanks to the Rayfield Library for making this advanced GUI possible."
})

-- **Lade gespeicherte Einstellungen**
Rayfield:LoadConfiguration()
