local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Hauptfenster erstellen
local Window = Rayfield:CreateWindow({
    Name = "Wow Effect Hub",
    LoadingTitle = "Loading Ultimate Experience",
    LoadingSubtitle = "Please wait while we prepare...",
    ConfigurationSaving = {
        Enabled = true,
        FolderName = "UltimateGUI", -- Ordner für gespeicherte Konfigurationen
        FileName = "WowHubConfig"
    },
    Discord = {
        Enabled = true,
        Invite = "example", -- Hier deine Discord-Einladung einfügen
        RememberJoins = true
    },
    Theme = "Serenity", -- Voreingestelltes Theme
    KeySystem = true,
    KeySettings = {
        Title = "Authentication Required",
        Subtitle = "Enter your access key to proceed.",
        Note = "Contact support if you don't have a key.",
        Key = "Your-Access-Key" -- Zugangsschlüssel einstellen
    }
})

-- Erster Tab: Hauptmenü
local MainTab = Window:CreateTab("Main Menu", 4483362458) -- Benutzerdefinierte Icon-ID

-- Einführungstext
MainTab:CreateLabel("Welcome to the Wow Effect Hub!")

-- Script ausführen
MainTab:CreateButton({
    Name = "Execute Cheese Script",
    Callback = function()
        -- Dein gewünschtes Script
        game:GetService("ReplicatedStorage").AddCheese:FireServer()
        print("Script executed successfully.")
    end,
})

-- Slider für Lautstärke
MainTab:CreateSlider({
    Name = "Adjust Volume",
    Range = {0, 100},
    Increment = 5,
    Suffix = "%",
    CurrentValue = 50,
    Callback = function(Value)
        print("Volume set to:", Value)
    end,
})

-- Dropdown für Modi
MainTab:CreateDropdown({
    Name = "Select Mode",
    Options = {"Stealth", "Combat", "Explorer"},
    CurrentOption = "Stealth",
    Callback = function(Option)
        print("Mode selected:", Option)
    end,
})

-- Zweiter Tab: Design und Themes
local DesignTab = Window:CreateTab("Design & Themes", 4483362458)

-- Theme-Wechsler
DesignTab:CreateDropdown({
    Name = "Pick a Theme",
    Options = {"Default", "AmberGlow", "Amethyst", "Ocean", "Serenity"},
    CurrentOption = "Serenity",
    Callback = function(Theme)
        Window:SetTheme(Theme)
        print("Theme changed to:", Theme)
    end,
})

-- Farbwähler mit Animation
DesignTab:CreateColorPicker({
    Name = "Customize Accent Color",
    Color = Color3.fromRGB(0, 255, 255),
    Callback = function(Color)
        Window:ModifyTheme({
            Accent = Color
        })
        print("Accent color changed:", Color)
    end,
})

-- Paragraf mit Infos
DesignTab:CreateParagraph({
    Title = "About This GUI",
    Content = "This GUI showcases the ultimate potential of Rayfield with full features and an impressive design."
})

-- Tabs laden und Konfiguration speichern
Rayfield:LoadConfiguration()
