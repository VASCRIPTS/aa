local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Fenster erstellen
local Window = Rayfield:CreateWindow({
    Name = "High-Class GUI",
    LoadingTitle = "High-Class Interface",
    LoadingSubtitle = "Initializing the best experience...",
    ConfigurationSaving = {
        Enabled = true,
        FolderName = "RayfieldProjects",
        FileName = "HighClassUI"
    },
    Discord = {
        Enabled = true,
        Invite = "example", -- Füge hier eine Discord-Einladung hinzu
        RememberJoins = true
    },
    Theme = "DarkBlue", -- Nutze ein passendes Theme (siehe Dokumentation)
    KeySystem = true,
    KeySettings = {
        Title = "Enter Your Key",
        Subtitle = "Access required to use this GUI",
        Note = "Contact support for access keys.",
        Key = "12345-ExampleKey" -- Ersetze dies mit deinem Schlüssel
    }
})

-- Haupt-Tab erstellen
local MainTab = Window:CreateTab("Dashboard", 4483362458)

-- Label hinzufügen
MainTab:CreateLabel("Welcome to the High-Class GUI")

-- Benachrichtigungsschaltfläche
MainTab:CreateButton({
    Name = "Show Welcome Notification",
    Callback = function()
        Rayfield:Notify({
            Title = "Welcome!",
            Content = "You have successfully opened the GUI.",
            Duration = 5,
            Actions = {
                Close = {
                    Name = "Close",
                    Callback = function()
                        print("Notification closed.")
                    end
                }
            }
        })
    end,
})

-- Slider für Lautstärke
MainTab:CreateSlider({
    Name = "Adjust Volume",
    Range = {0, 100},
    Increment = 10,
    Suffix = "%",
    CurrentValue = 50,
    Callback = function(Value)
        print("Volume set to:", Value)
    end,
})

-- Farbwähler
MainTab:CreateColorPicker({
    Name = "Pick a Theme Color",
    Color = Color3.fromRGB(0, 255, 255),
    Callback = function(Color)
        print("New color selected:", Color)
    end,
})

-- Dropdown-Menü
MainTab:CreateDropdown({
    Name = "Select Mode",
    Options = {"Mode 1", "Mode 2", "Mode 3"},
    CurrentOption = "Mode 1",
    Callback = function(Option)
        print("Selected Mode:", Option)
    end,
})

-- Eingabefeld
MainTab:CreateInput({
    Name = "Custom Command",
    PlaceholderText = "Type here...",
    RemoveTextAfterFocusLost = false,
    Callback = function(Text)
        print("Command entered:", Text)
    end,
})

-- Schalter (Toggle)
MainTab:CreateToggle({
    Name = "Activate Feature",
    CurrentValue = false,
    Callback = function(State)
        print("Feature activated:", State)
    end,
})

-- Keybind
MainTab:CreateKeybind({
    Name = "Quick Action Keybind",
    CurrentKeybind = Enum.KeyCode.F,
    HoldToInteract = false,
    Callback = function()
        print("Keybind activated!")
    end,
})

-- Paragraf hinzufügen
MainTab:CreateParagraph({
    Title = "Information",
    Content = "This GUI demonstrates the full potential of the Rayfield library. Explore and customize!"
})

-- Zusätzlicher Tab für Themes
local ThemeTab = Window:CreateTab("Themes", 4483362458)
ThemeTab:CreateLabel("Change the GUI Theme")
ThemeTab:CreateDropdown({
    Name = "Select Theme",
    Options = {"Default", "AmberGlow", "Amethyst", "Ocean", "Serenity"},
    CurrentOption = "DarkBlue",
    Callback = function(Theme)
        Window:ModifyTheme(Theme)
        print("Theme changed to:", Theme)
    end,
})

-- Tabs und Fenster laden
Rayfield:LoadConfiguration()
